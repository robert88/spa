  22. getCenter

    	$.dialog.fadeLeft=function($dialog){
			var center = $.dialog.getCenter($dialog)
			var w = $dialog.width();
			$dialog.css({left:-w,top:center.y}).stop(true,true).animate({left:center.x},200);
	}
	$(".trigger5").click(function(){
		$.dialog('hello world!',{animateType:'fadeLeft'});
